<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Simple JavaScript Blog</title>
    <style>
        body { font-family: Arial, sans-serif; max-width: 800px; margin: auto; padding: 20px; }
        h1 { text-align: center; }
        .post { border-bottom: 1px solid #ccc; padding: 10px 0; }
        .post h2 { margin: 0; }
        .post p { color: #555; }
    </style>
</head>
<body>
    <h1>My JavaScript Blog</h1>
    <div id="posts"></div>

    <script>
        // Sample blog posts stored in JSON format
        const posts = [
            { title: "First Blog Post", date: "2025-03-10", content: "This is my first blog post. Welcome!" },
            { title: "Why I Love JavaScript", date: "2025-03-11", content: "JavaScript is versatile and fun!" }
        ];

        // Function to display posts
        function loadPosts() {
            const postsContainer = document.getElementById("posts");
            posts.forEach(post => {
                const postElement = document.createElement("div");
                postElement.classList.add("post");
                postElement.innerHTML = `<h2>${post.title}</h2><p><i>${post.date}</i></p><p>${post.content}</p>`;
                postsContainer.appendChild(postElement);
            });
        }

        // Load posts when the page loads
        window.onload = loadPosts;
    </script>
</body>
</html>
